    ```julia
    """
        compute_photon_number(uomega, sim)

    Compute the photon number integral: N_ph = sum(|U(omega)|^2 / |omega + omega0|) * Delta_t.

    Uses carrier-offset absolute frequency (omega_s + omega_0) to avoid singularity
    at omega=0. This is the correct conserved quantity for the GNLSE with self-steepening
    (Agrawal NFF 6th ed. section 2.3).
    """
    function compute_photon_number(uomega, sim)
        omega_s = sim["ωs"]       # offset angular frequency grid (THz)
        omega_0 = sim["ω0"]       # carrier angular frequency (THz)
        Delta_t = sim["Δt"]       # time step (ps)
        # Carrier-offset avoids division by zero at DC bin
        abs_omega = abs.(omega_s .+ omega_0)
        return sum(abs2.(uomega) ./ abs_omega) * Delta_t
    end
    ```

    **VERIF-03 testset** (replaces placeholder):
    ```
    @testset "VERIF-03: Taylor remainder slope ~2 (adjoint O(eps^2) correct)" begin
        uomega0, fiber, sim, band_mask, _, _ = setup_raman_problem(
            Nt=VERIF_NT, L_fiber=0.5, P_cont=0.05, time_window=10.0,
            fiber_preset=:SMF28
        )

        # Random base phase and perturbation direction
        phi0 = 0.1 .* randn(VERIF_NT, 1)
        J0, grad = cost_and_gradient(phi0, uomega0, fiber, sim, band_mask)
        delta_phi = randn(VERIF_NT, 1)
        delta_phi ./= norm(delta_phi)
        directional_deriv = dot(vec(grad), vec(delta_phi))

        # Taylor remainder: |J(phi0 + eps*dphi) - J0 - eps*<grad, dphi>| should scale as eps^2
        epsilons = [1e-1, 1e-2, 1e-3, 1e-4]
        remainders = Float64[]
        for eps_val in epsilons
            J_eps, _ = cost_and_gradient(phi0 .+ eps_val .* delta_phi, uomega0, fiber, sim, band_mask)
            push!(remainders, abs(J_eps - J0 - eps_val * directional_deriv))
        end

        # Log-log slopes between consecutive epsilon pairs
        slopes = [log10(remainders[i] / remainders[i+1]) for i in 1:length(epsilons)-1]

        @info @sprintf("VERIF-03: Taylor remainder slopes: [%s]",
            join([@sprintf("%.2f", s) for s in slopes], ", "))
        @info @sprintf("VERIF-03: remainders: [%s]",
            join([@sprintf("%.2e", r) for r in remainders], ", "))

        # At least 2 of the 3 slopes should be in [1.4, 2.6] range
        # The first 2 pairs (1e-1->1e-2, 1e-2->1e-3) are most reliable;
        # the 3rd pair (1e-3->1e-4) may enter machine noise floor at Nt=2^14
        good_slopes = count(s -> 1.4 <= s <= 2.6, slopes)
        @test good_slopes >= 2

        push!(results, (name="VERIF-03: Taylor remainder",
            passed=good_slopes >= 2,
            evidence=@sprintf("slopes=[%s], %d/3 in [1.4,2.6]",
                join([@sprintf("%.2f", s) for s in slopes], ","), good_slopes)))
    end
    ```

    - Use `fiber_preset=:SMF28` and `L_fiber=0.5` (short fiber, same as existing test pattern)
