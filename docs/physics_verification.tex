\section{Notation and Conventions}

\begin{itemize}[leftmargin=*]
\item $u(z,t)$: slowly-varying electric field envelope in the co-moving frame, units $\sqrt{\mathrm{W}}$. Depends on propagation distance $z$ and retarded time $t$.
\item $\hat{u}(z,\omega)$: Fourier transform of $u(z,t)$ in time, units $\sqrt{\mathrm{W}}\cdot\mathrm{s}$.
\item $\tilde{u}(z,\omega)$: interaction-picture field, related to $\hat{u}$ by a dispersive phase (\S\ref{sec:interaction}).
\item $\omega_0 = 2\pi c / \lambda_0$: carrier angular frequency. For $\lambda_0 = 1550\,\mathrm{nm}$, $\omega_0 \approx 1.216 \times 10^{15}\,\mathrm{rad/s}$.
\item $\beta_n = \left.\dd[n]{\beta}{\omega}\right|_{\omega_0}$: $n$-th order dispersion coefficient of the propagation constant $\beta(\omega)$ about $\omega_0$.
\item $\gamma$: effective nonlinear coefficient, units $\mathrm{W}^{-1}\mathrm{m}^{-1}$.
\item $f_R$: fractional Raman contribution (dimensionless), $f_R = 0.18$ for silica.
\item $h_R(t)$: Raman response function, units $\mathrm{s}^{-1}$, causal ($h_R(t)=0$ for $t<0$).
\item $\mathcal{F}$, $\mathcal{F}^{-1}$: forward and inverse Fourier transforms in time.
\item $\ast$: temporal convolution.
\end{itemize}

\textbf{Code convention:} the code uses an inverted FFT sign convention. Specifically
$\hat{u} = \mathcal{F}^{-1}[u]$, so $u = \mathcal{F}[\hat{u}]$ in the code. This does
not change any physical equation --- it only affects where you write \code{fft} vs.\
\code{ifft}. Throughout this document, physics equations are written in the standard
convention; the code-convention note is only relevant when comparing the two.

\section{The Generalized Nonlinear Schr\"{o}dinger Equation}
\label{sec:gnlse}

\subsection{Lab-frame form}

The field envelope $u(z,t)$ propagates through a lossless single-mode fiber according
to the generalized nonlinear Schr\"{o}dinger equation (GNLSE) with dispersion, Kerr
nonlinearity, Raman response, and self-steepening. In the frequency domain it reads:
\begin{equation}
  \pdv{\hat{u}(z,\omega)}{z}
  = iD(\omega)\,\hat{u}(z,\omega)
  + i\gamma\,\frac{\omega}{\omega_0}\,
    \mathcal{F}\!\left[
      (1-f_R)\,|u|^2\,u \;+\; f_R\,\bigl(h_R \ast |u|^2\bigr)\,u
    \right].
  \label{eq:gnlse_lab}
\end{equation}

This is Agrawal, \textit{Nonlinear Fiber Optics} 6th ed., Eq.~(2.3.36), specialized
to the single-mode case. The left-hand side is the rate of change of the complex
spectrum along the fiber. The right-hand side has two pieces:
\begin{enumerate}[leftmargin=*]
\item \textbf{Linear dispersion} $iD(\omega)\hat{u}$, where
\begin{equation}
  D(\omega) \;=\; \sum_{n \ge 2} \frac{\beta_n}{n!}\,(\omega - \omega_0)^n,
  \label{eq:D_of_omega}
\end{equation}
excluding $\beta_0$ (absolute phase, unphysical) and $\beta_1$ (group velocity,
absorbed into the co-moving frame).

\item \textbf{Nonlinear response} with two sub-pieces: an instantaneous Kerr term
$(1-f_R)\,|u|^2 u$ and a delayed Raman term $f_R\,(h_R \ast |u|^2)\,u$, both scaled
by $\gamma$ and by the self-steepening factor $\omega/\omega_0$.
\end{enumerate}

\begin{verifybox}[Please verify]
(a) The dispersion operator excludes $\beta_0$ and $\beta_1$. \\
(b) The nonlinear response is Kerr$+$Raman with $f_R=0.18$ splitting. \\
(c) The self-steepening prefactor is $\omega/\omega_0$, not $(1+\omega/\omega_0)$.
This is equivalent to Agrawal's form under the carrier-absorbing transformation
and is the convention used throughout the code.
\end{verifybox}

\subsection{Interaction-picture form (what the code actually integrates)}
\label{sec:interaction}

The dispersion term $iD(\omega)\hat{u}$ is stiff: it produces rapid phase oscillations
that force the ODE solver to take tiny steps. We remove it analytically via the
interaction-picture transformation:
\begin{equation}
  \tilde{u}(z,\omega) \;=\; e^{-iD(\omega)\,z}\,\hat{u}(z,\omega).
  \label{eq:ip_transform}
\end{equation}

Substituting into Eq.~\eqref{eq:gnlse_lab} and simplifying, the linear term cancels
and we are left with the reduced ODE that the code integrates:
\begin{equation}
  \boxed{\;
    \dv{\tilde{u}(z,\omega)}{z}
    \;=\; i\,e^{-iD(\omega)\,z}\,\frac{\omega}{\omega_0}\,
      \mathcal{F}\!\left[
        \gamma\,(1-f_R)\,|u|^2 u
        \;+\; \gamma\,\bigl(h_R \ast |u|^2\bigr)\,u
      \right]\;}
  \label{eq:gnlse_ip}
\end{equation}
where $u(z,t)$ inside the brackets is the time-domain \emph{lab-frame} field,
reconstructed from $\tilde{u}$ via
\begin{equation}
  u(z,t) \;=\; \mathcal{F}^{-1}\!\Bigl[\,e^{+iD(\omega)\,z}\,\tilde{u}(z,\omega)\,\Bigr].
  \label{eq:ip_inverse}
\end{equation}

\paragraph{Why the interaction picture helps.} The linear dispersion causes frequency
components to acquire phase at the rate $D(\omega)$. For typical SMF-28 over 1\,m
of propagation, $D(\omega)L \sim 100$ radians at the edge of the signal band. A
direct integrator would need thousands of sub-steps to resolve this phase. In the
interaction picture, the linear phase is handled analytically by
Eq.~\eqref{eq:ip_inverse}, and the solver only needs to resolve the much slower
nonlinear dynamics --- typically hundreds of steps per meter instead of millions.

\subsection{Code implementation}

The reduced ODE Eq.~\eqref{eq:gnlse_ip} is the right-hand side of the ODE solver.
The function is \code{disp\_mmf!} in \file{src/simulation/simulate\_disp\_mmf.jl},
lines 25--61.

\begin{lstlisting}[caption={\file{src/simulation/simulate\_disp\_mmf.jl} L25--L61},firstnumber=25]
function disp_mmf!(dũω, ũω, p, z)
    selfsteep, Dω, γ, hRω, one_m_fR, attenuator, ..., = p

    @. exp_D_p = cis(Dω * z)                        # L28:  e^{+iDz}
    @. exp_D_m = cis(-Dω * z)                       # L29:  e^{-iDz}

    @. uω = exp_D_p * ũω                            # L31:  lab frame (Eq. 4)

    fft_plan_M! * uω                                # L33:  -> time domain
    @. ut = attenuator * uω                         # L34:  boundary window
    @. v = real(ut)
    @. w = imag(ut)

    # Kerr: η_K = (1-fR) γ |u|^2 u
    @tullio δKt[t,i,j] = γ[i,j,k,l]*(v[t,k]*v[t,l] + w[t,k]*w[t,l])   # L39
    @tullio αK[t,i] = δKt[t,i,j] * v[t,j]           # L40
    @tullio βK[t,i] = δKt[t,i,j] * w[t,j]           # L41
    @. ηKt = αK + 1im * βK                          # L42
    @. ηKt *= one_m_fR                              # L43:  (1-fR) factor

    # Raman: η_R = γ (h_R * |u|^2) u
    @. δKt_cplx = ComplexF64(δKt, 0.0)              # L46
    fft_plan_MM! * δKt_cplx                         # L47:  -> freq for convolution
    @. hRω_δRω = hRω * δKt_cplx                     # L48:  freq-domain convolution
    ifft_plan_MM! * hRω_δRω                         # L49:  -> back to time
    fftshift!(hR_conv_δR, hRω_δRω, 1)               # L50
    @. δRt = real(hR_conv_δR)
    @tullio αR[t,i] = δRt[t,i,j] * v[t,j]           # L52
    @tullio βR[t,i] = δRt[t,i,j] * w[t,j]           # L53
    @. ηRt = αR + 1im * βR

    @. ηt = ηKt + ηRt                               # L57:  Kerr + Raman
    ifft_plan_M! * ηt                               # L58:  -> freq
    ηt .*= selfsteep                                # L59:  ω/ω₀ factor
    @. dũω = 1im * exp_D_m * ηt                     # L60:  i * e^{-iDz} * (...)
end
\end{lstlisting}

Line-by-line correspondence:
\begin{itemize}[leftmargin=*]
\item \line{28--29}: the two phase factors $e^{\pm iDz}$ from Eqs.~\eqref{eq:ip_transform}, \eqref{eq:ip_inverse}.
\item \line{31}: inverse interaction-picture transform, Eq.~\eqref{eq:ip_inverse} in the frequency domain.
\item \line{33}: FFT to time to evaluate the nonlinear response pointwise.
\item \line{39}: Kerr operator $\delta_K = \gamma\,|u|^2$ via the contraction $\gamma_{ijkl}(v_k v_l + w_k w_l)$ with $v = \Re u$, $w = \Im u$.
\item \line{43}: factor $(1-f_R)$ on the Kerr term (this is the $(1-f_R)$ in Eq.~\eqref{eq:gnlse_ip}).
\item \line{48}: frequency-domain Raman convolution $\widehat{h_R}\cdot\widehat{|u|^2}$, which becomes $h_R\ast|u|^2$ in time.
\item \line{59}: self-steepening factor $\omega/\omega_0$ (the array \code{selfsteep} is built once at setup from $\omega_s/\omega_0$).
\item \line{60}: the overall factor $i\,e^{-iDz}$ of Eq.~\eqref{eq:gnlse_ip}.

\section{Dispersion Operator}
\label{sec:dispersion}

The dispersion operator $D(\omega)$ is the Taylor expansion of the propagation
constant $\beta(\omega)$ about $\omega_0$, with $\beta_0$ and $\beta_1$ removed:
\begin{equation}
  D(\omega)
  \;=\; \beta(\omega) - \beta(\omega_0) - \beta_1\,(\omega - \omega_0)
  \;=\; \sum_{n=2}^{N_\beta}\frac{\beta_n}{n!}\,(\omega - \omega_0)^n.
  \label{eq:D_taylor}
\end{equation}
In the code, we set $\Omega \equiv \omega - \omega_0$ and work with the FFT-domain
angular frequency grid $\Omega = 2\pi\,\Delta f$.

\paragraph{Fiber parameters used.} The two presets are defined in
\file{scripts/common.jl}, lines 47--76:

\midrule
\code{:SMF28}  & $1.1 \times 10^{-3}$  & $(-2.17\times 10^{-26},\;1.2\times 10^{-40})$ \\
\code{:HNLF}   & $1.0 \times 10^{-2}$  & $(-5.0\times 10^{-27},\;1.0\times 10^{-40})$ \\
\bottomrule
\end{tabular}
\end{center}

\paragraph{Operator construction.} The dispersion operator is built once at setup in
\code{get\_disp\_fiber\_params\_user\_defined}, \file{src/helpers/helpers.jl},
lines 161--193. Line 187 is the operative line:
\begin{lstlisting}[caption={\file{src/helpers/helpers.jl} L185--L188},firstnumber=185]

βn_ω   = reshape(vcat([0.0, 0.0], β_tail), :, 1)        # [0, 0, β₂, β₃, ...]

\section{Raman Response Function}
\label{sec:raman}

The delayed Raman response is modeled by the Blow--Wood single-oscillator form
(Agrawal Eq.~2.3.40):
\begin{equation}
  h_R(t) \;=\; \frac{\tau_1^2 + \tau_2^2}{\tau_1\,\tau_2^2}\,
               e^{-t/\tau_2}\,\sin\!\left(\frac{t}{\tau_1}\right)\,\Theta(t),
  \label{eq:hR_blow_wood}
\end{equation}
where $\Theta(t)$ is the Heaviside step function (enforcing causality: the glass
cannot respond before the excitation arrives), and the time constants for fused
silica are
\begin{equation}
  \tau_1 = 12.2\,\mathrm{fs}, \qquad \tau_2 = 32\,\mathrm{fs}.
\end{equation}
The factor $(\tau_1^2+\tau_2^2)/(\tau_1\tau_2^2)$ is a normalization such that
$\int_0^\infty h_R(t)\,dt = 1$.

\paragraph{Code implementation.} \file{src/helpers/helpers.jl}, lines 181--183
(for user-defined single-mode fibers; there is a structurally identical copy at
L106--L108 for the GRIN path):
\begin{lstlisting}[caption={\file{src/helpers/helpers.jl} L181--L183},firstnumber=181]
ts_pos = max.(ts, 0.0)
hRt = fR * Δt * 1e3 * (τ1^2 + τ2^2) / (τ1 * τ2^2) .*
      exp.(-ts_pos * 1e15 / τ2) .* sin.(ts_pos * 1e15 / τ1) .*
      (sign.(ts) .+ 1) / 2
hRω = fft(hRt)
\end{lstlisting}

Two things to notice:
\begin{itemize}[leftmargin=*]
\item The factor \code{(sign(ts) + 1)/2} is the Heaviside step $\Theta(t)$. It equals $1$ for $t>0$, $0$ for $t<0$, and $1/2$ at $t=0$.
\item The \code{max(ts, 0.0)} on line 181 is a numerical-stability guard, not physics. Without it, the \code{exp(-t/$\tau_2$)} factor would overflow \code{Float64} for negative $t$ (where $e^{|t|/\tau_2}$ diverges) even though the Heaviside would then multiply it by zero. Pre-clamping $t$ to $\max(t,0)$ makes the intermediate quantity finite. The result is mathematically identical to Eq.~\eqref{eq:hR_blow_wood}.
\end{itemize}

\begin{verifybox}[Please verify]
(a) The Blow--Wood form matches your expectation for $h_R(t)$ in silica. \\
(b) The Heaviside factor is correctly present (causality). \\
(c) The $\max(t,0)$ clamp is a numerical artifact only; it does not change
any non-zero value of $h_R$.
\end{verifybox}

\section{Initial Pulse Shape}
\label{sec:initial_pulse}

All simulations use a sech$^2$ pulse at $\lambda_0 = 1550\,\mathrm{nm}$ with
$\mathrm{FWHM} = 185\,\mathrm{fs}$. The time-domain envelope is
\begin{equation}
  u(0,t) \;=\; \sqrt{P_\text{peak}}\,\sech\!\left(\frac{t}{T_0}\right),
  \qquad T_0 = \frac{\mathrm{FWHM}}{1.7627},
  \label{eq:sech_pulse}
\end{equation}
where $1.7627 = 2\,\operatorname{arcosh}\sqrt{2}$. The intensity
$|u|^2 = P_\text{peak}\sech^2(t/T_0)$ has the desired full-width at half-maximum.

The simulation works with average power $P_\text{cont}$ (easier to specify
experimentally). The conversion is
\begin{equation}
  P_\text{peak} \;=\; 0.881374 \times \frac{P_\text{cont}}{\mathrm{FWHM}\,\cdot\,f_\text{rep}},
  \label{eq:peak_power}
\end{equation}
where $0.881374 = \int_{-\infty}^\infty \sech^2(x)\,dx \,/\,(2\,\operatorname{arcosh}\sqrt{2})$
is the sech$^2$ shape factor for converting peak-to-average.

\paragraph{Code implementation.} \file{src/simulation/simulate\_disp\_mmf.jl},
lines 135--152:

\section{Cost Functional (Physical Interpretation)}
\label{sec:cost}

The optimization minimizes the fractional spectral energy in a defined Raman band:
\begin{equation}
  J \;=\; \frac{E_\text{band}}{E_\text{total}}
      \;=\; \frac{\displaystyle\int_\text{band} |\hat{u}(L,\omega)|^2\,d\omega}
                  {\displaystyle\int_{-\infty}^{\infty}|\hat{u}(L,\omega)|^2\,d\omega}
  \;\in\;[0,1].
  \label{eq:cost_continuous}
\end{equation}
The ``band'' is defined as all FFT bins with frequency offset $\Delta f < -13\,\mathrm{THz}$
relative to the carrier, which captures the Stokes-shifted Raman content. The
definition is the same for both fibers.

On the discrete grid, this becomes
\begin{equation}
  J \;=\; \frac{\sum_{k \in \mathcal{B}} |\hat{u}_k|^2}
                {\sum_{k=1}^{N_t} |\hat{u}_k|^2},
  \label{eq:cost_discrete}
\end{equation}
where $\mathcal{B}$ is the set of spectral bins in the Raman band.

\paragraph{Code implementation.} \file{scripts/common.jl}, lines 260--276:
\begin{lstlisting}[caption={\file{scripts/common.jl} L260--L276},firstnumber=260]
function spectral_band_cost(uωf, band_mask)
    @assert size(uωf,1) == length(band_mask)
    @assert any(band_mask)
    @assert sum(abs2.(uωf)) > 0

    E_band  = sum(abs2.(uωf[band_mask, :]))        # L266
    E_total = sum(abs2.(uωf))                      # L267
    J       = E_band / E_total                     # L268
    dJ      = uωf .* (band_mask .- J) ./ E_total   # L269: terminal adjoint

    @assert 0 ≤ J ≤ 1
    @assert all(isfinite, dJ)
    return J, dJ
end
\end{lstlisting}

The second return value \code{dJ} is the terminal condition for the adjoint
sensitivity equation; it is the derivative of $J$ with respect to $\hat{u}^*(L,\omega)$,
which you can derive by differentiating Eq.~\eqref{eq:cost_discrete}.

\begin{verifybox}[Please verify]
(a) The cost is energy-fractional and dimensionless, bounded in $[0,1]$. \\
(b) The Raman band threshold of $-13\,\mathrm{THz}$ correctly captures
the Stokes shift in silica. \\
(c) The gradient expression on line 269 is the correct derivative
$\partial J / \partial \hat{u}^*(L)$.
\end{verifybox}

\section{Forward Propagation Solver}
\label{sec:solver}

The ODE Eq.~\eqref{eq:gnlse_ip} is integrated from $z = 0$ to $z = L$ using
Julia's \code{DifferentialEquations.jl}. We use \code{Tsit5} (Tsitouras'
5th-order Runge--Kutta method) with relative tolerance $10^{-8}$.

\file{src/simulation/simulate\_disp\_mmf.jl}, lines 176--198:
\begin{lstlisting}[caption={\file{src/simulation/simulate\_disp\_mmf.jl} L176--L198},firstnumber=176]
function solve_disp_mmf(uω0, fiber, sim)
    p_disp_mmf = get_p_disp_mmf(..., sim["attenuator"])                # L177
    prob_disp_mmf = ODEProblem(disp_mmf!, uω0, (0, fiber["L"]), p_disp_mmf)  # L179

    if isnothing(fiber["zsave"])
        sol = solve(prob_disp_mmf, Tsit5(), reltol=1e-8)               # L182
        return Dict("ode_sol" => sol)
    else
        sol = solve(prob_disp_mmf, Tsit5(), reltol=1e-8,
                     saveat=fiber["zsave"])                            # L186
        uω_z = zeros(ComplexF64, length(fiber["zsave"]), sim["Nt"], sim["M"])
        ut_z = zeros(ComplexF64, length(fiber["zsave"]), sim["Nt"], sim["M"])
        for i in 1:length(fiber["zsave"])
            uω_z[i,:,:] = cis.(fiber["Dω"]*fiber["zsave"][i]) .* sol(fiber["zsave"][i])
            ut_z[i,:,:] = fft(uω_z[i,:,:], 1)
        end
        return Dict("ode_sol"=>sol, "uω_z"=>uω_z, "ut_z"=>ut_z)
    end
end
\end{lstlisting}

Line 192 converts the interaction-picture field $\tilde{u}$ back to the lab frame
by multiplying by $e^{+iD(\omega)z}$, which is the inverse of the transformation in
Eq.~\eqref{eq:ip_transform}.

\paragraph{Boundary condition: super-Gaussian window.} To prevent FFT wraparound,
a super-Gaussian window of order 30 is applied to the time-domain field on every
RHS evaluation (line 34 of \code{disp\_mmf!}). This is a soft absorbing boundary
condition --- it attenuates field content that approaches the edge of the time
window. The window is built in \code{get\_disp\_sim\_params},
\file{src/helpers/helpers.jl} lines 59--63:
\begin{lstlisting}[caption={\file{src/helpers/helpers.jl} L59--L63},firstnumber=59]

n_attenuation = 30                              # super-Gaussian order
σ_attenuation = r_attenuation / log(2)^(1/n_attenuation)
...
attenuator = exp.(-(abs.(1e12*ts)/σ_attenuation).^n_attenuation) * ones(M)'
\end{lstlisting}
The functional form is
\begin{equation}
  w(t) \;=\; \exp\!\left[-\left(\frac{|t|}{\sigma}\right)^{30}\right],
  \qquad \sigma \;=\; \frac{0.85\,W/2}{(\ln 2)^{1/30}},
\end{equation}
where $W$ is the full time window. For $n=30$, $w(t)\approx 1$ for
$|t|<0.85\,W/2$ and drops sharply to zero near $|t|=W/2$.

\begin{verifybox}[Please verify]
(a) \code{Tsit5} with \code{reltol=1e-8} is an acceptable solver for this class
of nonlinear Schr\"odinger equation at the soliton numbers we study ($N \le 6$). \\
(b) The super-Gaussian boundary treatment does not interfere with the pulse
physics as long as the time window is sized so that the pulse never reaches
$|t|=0.85\,W/2$. This is checked by the photon-conservation test
(\S\ref{sec:verif_photon}).
\end{verifybox}

\section{Time Window Sizing}
\label{sec:timewindow}

The time window $W$ must be large enough that the pulse does not reach the
absorbing boundary at any point during propagation. Two effects cause the pulse
to spread in time:
\begin{enumerate}[leftmargin=*]
\item \textbf{Dispersive walk-off} between frequencies separated by the Raman shift
$\Delta\omega_R = 2\pi \times 13\,\mathrm{THz}$. Different frequencies travel at
different group velocities, so a pair of points separated by $\Delta\omega_R$
walk apart by
\begin{equation}
  \Delta t_\text{walk} \;=\; |\beta_2|\,L\,\Delta\omega_R.
  \label{eq:walk_off}
\end{equation}

\item \textbf{SPM-induced spectral broadening} converted back into temporal spread
via GVD. The nonlinear phase accumulated over the fiber is
\begin{equation}
  \phi_\text{NL} \;=\; \gamma\,P_\text{peak}\,L \qquad \text{(radians, dimensionless)}.
  \label{eq:phi_NL}
\end{equation}
The SPM-broadened angular bandwidth of a sech$^2$ pulse is (Agrawal Ch. 4)
\begin{equation}
  \Delta\omega_\text{SPM} \;=\; 0.86\,\frac{\phi_\text{NL}}{T_0}.
  \label{eq:delta_omega_spm}
\end{equation}
Propagating this broadened spectrum through GVD produces an additional temporal
spread of
\begin{equation}
  \Delta t_\text{SPM} \;=\; |\beta_2|\,L\,\Delta\omega_\text{SPM}.
  \label{eq:delta_t_spm}
\end{equation}
\end{enumerate}

The recommended time window adds both terms with a safety factor:
\begin{equation}
  W \;\gtrsim\; 2\,\bigl(\,\Delta t_\text{walk} + \Delta t_\text{SPM} + 0.5\,\mathrm{ps}\,\bigr).
  \label{eq:W_recommended}
\end{equation}

\paragraph{Critical note --- dimensional analysis.} It is tempting to try to use
$\phi_\text{NL}$ directly as a frequency scale, but $\phi_\text{NL}$ is
\emph{dimensionless} (radians). The division by $T_0$ in
Eq.~\eqref{eq:delta_omega_spm} is essential: $\mathrm{rad}/\mathrm{s} =
\mathrm{rad}\,/\,\mathrm{s}$. An earlier version of the code was missing this
$1/T_0$ factor and produced time windows 5--7$\times$ too small at high power,
which caused 38--49\% photon number loss at the absorbing boundary. This was
caught by the photon-conservation test (\S\ref{sec:verif_photon}) and fixed.

\paragraph{Code implementation.} \file{scripts/common.jl}, lines 191--215:
\begin{lstlisting}[caption={\file{scripts/common.jl} L191--L215},firstnumber=191]
function recommended_time_window(L_fiber; safety_factor=2.0, beta2=20e-27,
                                 gamma=0.0, P_peak=0.0, pulse_fwhm=185e-15)
    @assert L_fiber > 0
    @assert beta2 > 0
    @assert pulse_fwhm > 0

    Δω_raman   = 2π * 13e12                              # L198: 2π × 13 THz
    walk_off_ps = beta2 * L_fiber * Δω_raman * 1e12      # L199: Eq. (19)
    pulse_extent = 0.5

    spm_ps = 0.0
    if gamma > 0 && P_peak > 0
        T0     = pulse_fwhm / 1.763                       # sech^2 half-duration [s]
        φ_NL   = gamma * P_peak * L_fiber                 # L208: Eq. (20) [rad]
        δω_SPM = 0.86 * φ_NL / T0                         # L209: Eq. (21) [rad/s]
        spm_ps = beta2 * L_fiber * δω_SPM * 1e12          # L210: Eq. (22) [ps]
    end

    total_ps = walk_off_ps + spm_ps + pulse_extent
    return max(5, ceil(Int, total_ps * safety_factor))   # L214: Eq. (23)
end
\end{lstlisting}

\begin{verifybox}[Please verify]
(a) The walk-off formula on line 199 corresponds to Eq.~\eqref{eq:walk_off}.\\
(b) Line 209 implements Eq.~\eqref{eq:delta_omega_spm}, including the
dimensional $1/T_0$ factor. \\
(c) The safety factor of 2 in Eq.~\eqref{eq:W_recommended} is sufficient for
your range of intended operating conditions.
\end{verifybox}

\section{Adjoint Sensitivity Equation}
\label{sec:adjoint}

To optimize the input spectral phase efficiently, we use the adjoint method to
compute the gradient $\partial J / \partial \varphi$ with only \emph{one}
additional backward propagation, instead of the $N_t$ finite-difference
perturbations that na\"ive differentiation would require.

\subsection{Derivation (sketch)}

Start from the optimization problem
\begin{equation}
  \min_{\varphi(\omega)}\; J\bigl(\hat{u}(L)\bigr)
  \quad\text{subject to}\quad
  \pdv{\hat{u}}{z} = F\bigl(\hat{u},z\bigr),
  \quad
  \hat{u}(0,\omega) \;=\; \hat{u}_0(\omega)\,e^{i\varphi(\omega)},
\end{equation}
where $F$ is the GNLSE right-hand side (Eq.~\eqref{eq:gnlse_lab}). Introduce a
Lagrange multiplier field $\lambda(z,\omega)$ --- the adjoint field --- and write
the Lagrangian
\begin{equation}
  \mathcal{L}
  \;=\; J\bigl(\hat{u}(L)\bigr)
  \;+\; \int_0^L \Re\!\left\langle \lambda,\;
                   \pdv{\hat{u}}{z} - F(\hat{u},z)\right\rangle\,dz.
\end{equation}
Stationarity of $\mathcal{L}$ with respect to $\hat{u}(z)$ (after integration by
parts in $z$) gives the \textbf{backward adjoint equation}
\begin{equation}
  \boxed{\;
    -\dv{\lambda(z,\omega)}{z}
    \;=\; \left[\pdv{F}{\hat{u}}\right]^\dagger \lambda(z,\omega)
    \;}
  \label{eq:adjoint_ode}
\end{equation}
with the \textbf{terminal condition}
\begin{equation}
  \lambda(L,\omega)
  \;=\; \pdv{J}{\hat{u}^*}\bigg|_{z=L}.
  \label{eq:adjoint_terminal}
\end{equation}

The $\dagger$ denotes the adjoint operator (conjugate transpose in the
discretized problem). For our cost Eq.~\eqref{eq:cost_discrete}, the terminal
condition evaluates to
\begin{equation}
  \lambda(L,\omega)
  \;=\; \frac{\hat{u}(L,\omega)\bigl(\mathbb{1}_\text{band}(\omega) - J\bigr)}
              {E_\text{total}},
  \label{eq:terminal_expression}
\end{equation}
which is exactly what \code{spectral\_band\_cost} returns as \code{dJ} on
line 269 of \file{scripts/common.jl}.

\subsection{Gradient with respect to the input phase}

Once $\lambda(0,\omega)$ is known from backward integration, the gradient of
$J$ with respect to the input phase $\varphi(\omega)$ comes from the chain rule
through $\hat{u}(0,\omega) = \hat{u}_0(\omega)\,e^{i\varphi}$:
\begin{equation}
  \pdv{J}{\varphi(\omega)}
  \;=\; 2\,\Re\!\left[\,i\,\hat{u}(0,\omega)\,\lambda^*(0,\omega)\,\right].
  \label{eq:phase_gradient}
\end{equation}

\subsection{Code implementation}

The adjoint ODE right-hand side is \code{adjoint\_disp\_mmf!} in
\file{src/simulation/sensitivity\_disp\_mmf.jl}, lines 32--79. It has the same
overall structure as the forward solver but computes a linearized-transposed
operator. The four physical pieces are:
\begin{itemize}[leftmargin=*]
\item \line{72}: Kerr+Raman $\delta_1$ contraction ($|u|^2 u$ linearization)
\item \line{73}: conjugate Kerr $\delta_2$ contraction ($u^2$ linearization)
\item \line{74}: forward-path Raman convolution with $h_R$
\item \line{75}: conjugate-path Raman convolution with $h_R^*$
\end{itemize}
which are summed on line 78 to produce the adjoint time-derivative. Each piece
is the formal transpose of the corresponding forward term in \code{disp\_mmf!}.

\paragraph{Numerical accuracy.} The forward solver uses \code{Tsit5} with
continuous interpolation. When the adjoint queries the forward field at an
arbitrary $z$ (via \code{\~{}\textomega}(z) on line 50), it is evaluated through
\code{Tsit5}'s 4th-order Hermite interpolant. Consequently the adjoint's
accuracy is bounded by that 4th-order interpolant, and using a higher-order
backward solver would not improve anything. We use \code{Tsit5} with
\code{reltol=1e-8} for the adjoint as well. This choice is validated by the
Taylor-remainder test (\S\ref{sec:verif_taylor}).

\begin{verifybox}[Please verify]
(a) The adjoint sensitivity equation Eq.~\eqref{eq:adjoint_ode} is the
standard form for the backward sensitivity of an ODE-constrained optimization
problem. \\
(b) The terminal condition Eq.~\eqref{eq:adjoint_terminal} matches
Eq.~\eqref{eq:terminal_expression} for our cost functional. \\
(c) The phase-gradient formula Eq.~\eqref{eq:phase_gradient} is the chain rule
through the input phase mask.
\end{verifybox}

\section{Verification Tests}
\label{sec:verification}

We run four independent physics tests at the production grid size $N_t = 2^{14}$
in \file{scripts/verification.jl}. Each is derived from a physical invariant or
analytical solution that must hold if the code is correct.

\subsection{VERIF-01: Soliton shape preservation}
\label{sec:verif_soliton}

\textbf{Physical principle.} The fundamental $N=1$ soliton is an exact analytical
solution of the standard NLSE (Kerr nonlinearity with anomalous dispersion, no
Raman):
\begin{equation}
  u(z,t) \;=\; \sqrt{P_\text{peak}}\,\sech\!\left(\frac{t}{T_0}\right)
               \exp\!\left[\,i\,\frac{z}{2\,L_D}\,\right],
  \qquad
  P_\text{peak} \;=\; \frac{|\beta_2|}{\gamma\,T_0^2}.
  \label{eq:fundamental_soliton}
\end{equation}
After one soliton period $z_\text{sol} = (\pi/2)\,L_D$ (with
$L_D = T_0^2/|\beta_2|$), the pulse returns to its original shape. Any deviation
is numerical error.

\textbf{Code implementation.} \file{scripts/verification.jl} lines 73--150.
The test runs at $\beta_2 = -2.6\times 10^{-26}$ s$^2$/m, $\gamma = 1.3\times 10^{-3}$
W$^{-1}$m$^{-1}$, $T_0 = 185\,\mathrm{fs}/(2\operatorname{arcosh}\sqrt{2})$, with
$f_R$ set to $10^{-15}$ (Raman effectively disabled).

\textbf{Pass criterion:} $\max |I_\text{out}(t)/\max I_\text{out} - I_\text{in}(t)/\max I_\text{in}| < 0.02$ (2\%) across the pulse core (intensity $>$5\% of peak).

\textbf{Last result:} max deviation $= 0.0112$ (\textbf{PASS}, 1.1\% $<$ 2\%).

\subsection{VERIF-02: Photon number conservation}
\label{sec:verif_photon}

\textbf{Physical principle.} In a lossless fiber, Raman scattering transfers
\emph{energy} from higher to lower frequencies by exciting phonons, so
classical energy $\int|\hat{u}|^2\,d\omega$ is \emph{not} conserved. But every
Raman event takes one photon at frequency $\omega_1$ and produces one photon at
$\omega_2 < \omega_1$, so the \emph{photon number}
\begin{equation}
  N_\text{ph}
  \;=\; \int \frac{|\hat{u}(\omega)|^2}{\hbar\,\omega}\,d\omega
  \;\propto\; \int \frac{|\hat{u}(\omega)|^2}{\omega}\,d\omega
  \label{eq:photon_number}
\end{equation}
is conserved. This is the correct invariant to use for verifying a GNLSE with
self-steepening and Raman (Agrawal Ch.~2.3; Brabec \& Krausz, RMP 2000).

\textbf{Code implementation.} \file{scripts/verification.jl} lines 188--199
(\code{compute\_photon\_number}) and lines 211--252 (test loop over 5 production
configurations). The $\hbar$ constant is dropped since we check the \emph{ratio}
$N_\text{ph}(L)/N_\text{ph}(0)$.

\textbf{Pass criterion:} $|N_\text{ph}(L)/N_\text{ph}(0) - 1| < 0.01$ (1\%) for all 5 runs.

\textbf{Last result (post SPM-formula fix):} drifts $<$ 1\% on all 5 configurations
(\textbf{PASS}). Prior to the fix, high-power runs showed up to 49\% drift, which
led directly to discovering and fixing the time-window dimensional error in
\S\ref{sec:timewindow}.

\begin{verifybox}[Please verify]
(a) Photon number, not energy, is the correct conservation law for the GNLSE
with Raman (energy goes into phonons at each Raman event). \\
(b) The 1\% tolerance is acceptable given the super-Gaussian boundary window,
which is a physical approximation to an open boundary. \\
(c) The post-fix drifts are at the level of numerical solver tolerance plus
attenuator leakage, not a physical error.
\end{verifybox}

\subsection{VERIF-03: Taylor remainder for the adjoint gradient}
\label{sec:verif_taylor}

\textbf{Physical principle.} If $\nabla J$ is the correct gradient of $J$, then
Taylor's theorem guarantees that
\begin{equation}
  R(\epsilon)
  \;\equiv\; \bigl|\,J(\varphi + \epsilon\,\delta\varphi)
               - J(\varphi)
               - \epsilon\,\langle \nabla J,\,\delta\varphi\rangle\,\bigr|
  \;=\; O(\epsilon^2).
  \label{eq:taylor_remainder}
\end{equation}
Plotting $\log R(\epsilon)$ versus $\log\epsilon$ gives a straight line of
\emph{slope $2$} if the gradient is correct. A slope of $1$ signals a linear
bias in the gradient (an error proportional to $\delta\varphi$); a slope of $0$
signals a zero-order bias. Intermediate slopes near $2$ (say, in $[1.7,2.3]$)
are treated as passing.

\textbf{Code implementation.} \file{scripts/verification.jl} lines 271--330.
Runs at $L_\text{fiber} = 0.1$ m (short enough that higher-order Taylor terms
are negligible). Computes $R$ at $\epsilon \in \{1, 10^{-1}, 10^{-2}, 10^{-3}\}$.

\textbf{Pass criterion:} slopes between consecutive $\epsilon$ pairs all in
$[1.4, 2.6]$.

\textbf{Last result:} slopes $= [2.01,\,2.07,\,2.09]$, all in range
(\textbf{PASS}). This is strong evidence that the hand-derived adjoint of
\S\ref{sec:adjoint} has no sign error, no missing factor, and no off-by-one in
the chain rule.

\subsection{VERIF-04: Cost function cross-check}
\label{sec:verif_cost}

\textbf{Physical principle.} The cost $J$ is a simple ratio of integrals
(Eq.~\eqref{eq:cost_discrete}), so the result of the optimized routine
\code{spectral\_band\_cost} must match a direct evaluation of
$E_\text{band}/E_\text{total}$ to machine precision. This tests that the
band-mask indexing is correct and that there are no \code{fftshift} confusions.

\textbf{Last result:} $|J_\text{func} - J_\text{direct}| = 0$ exactly
(\textbf{PASS} at $10^{-12}$ tolerance).

\section{What the Tests Do and Do Not Prove}

\textbf{The tests prove:}
\begin{itemize}[leftmargin=*]
\item Given the model (Eq.~\eqref{eq:gnlse_ip}) and the parameter values
(\S\ref{sec:dispersion}), the code solves the equations correctly.
\item The forward propagation reproduces an exact analytical soliton to $1.1\%$.
\item Photon number is conserved to $<1\%$ across all tested operating points.
\item The adjoint gradient has no first-order bias (Taylor slope $\approx 2$).
\item The cost function is computed as claimed.
\end{itemize}

\textbf{The tests do not prove:}
\begin{itemize}[leftmargin=*]
\item That the physics model is appropriate for the experimental regime you
care about.
\item That $\gamma$, $\beta_2$, $\beta_3$ for the specific fibers in your lab
match the presets in \S\ref{sec:dispersion}.
\item That the optimizer finds the global minimum of Eq.~\eqref{eq:cost_discrete};
only that whatever local minimum it finds is computed faithfully.
\end{itemize}
