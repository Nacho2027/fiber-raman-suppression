**Section 3: Load results from manifest + JLD2**

Follow Research "Manifest Discovery and Loading Pattern":
```julia
manifest_path = joinpath("results", "raman", "manifest.json")
@assert isfile(manifest_path) "manifest.json not found — re-run section 2"
manifest = JSON3.read(read(manifest_path, String), Vector{Dict{String,Any}})

all_runs = Dict{String,Any}[]
for entry in manifest
    jld2_path = entry["result_file"]
    if !isfile(jld2_path)
        @warn "Missing JLD2 file" path=jld2_path
        continue
    end
    data = merge(Dict{String,Any}(entry), JLD2.load(jld2_path))
    push!(all_runs, data)
end
@info "Loaded $(length(all_runs)) runs from manifest"
```

**Section 4: Compute soliton number and update manifest (PATT-02 / D-05)**

For each run:
```julia
for run in all_runs
    betas = run["betas"]
    beta2 = isempty(betas) ? (run["fiber_name"] == "SMF-28" ? SMF28_BETAS[1] : HNLF_BETAS[1]) : betas[1]
    N = compute_soliton_number(Float64(run["gamma"]), Float64(run["P_cont_W"]),
                                Float64(run["fwhm_fs"]), Float64(beta2))
    run["soliton_number_N"] = N
    @info @sprintf("  %s L=%.1fm P=%.2fW -> N = %.2f", run["fiber_name"], run["L_m"], run["P_cont_W"], N)
end
```

Update manifest.json with soliton numbers:
```julia
for (i, entry) in enumerate(manifest)
```
