```julia
function compute_photon_number(uomega, sim)
    omega_s = sim["ωs"]
    Delta_t = sim["Δt"]
    abs_omega = abs.(omega_s)
    return sum(abs2.(uomega) ./ abs_omega) * Delta_t
end
```

From scripts/visualization.jl (existing helper):
```julia
function compute_soliton_number(gamma_Wm, P0_W, fwhm_fs, beta2_s2m)
    T0_s = (fwhm_fs * 1e-15) / (2.0 * acosh(sqrt(2.0)))
    N_sq = gamma_Wm * P0_W * T0_s^2 / abs(beta2_s2m)
    return sqrt(max(N_sq, 0.0))
end
```
