**Helper function: `pz_load_and_repropagate`**
```julia
function pz_load_and_repropagate(fiber_dir, config_name, preset; n_zsave=PZ_N_ZSAVE)
    jld2_path = joinpath("results", "raman", "sweeps", fiber_dir, config_name, "opt_result.jld2")
    data = JLD2.load(jld2_path)
    phi_opt = vec(data["phi_opt"])
    L = Float64(data["L_m"])
    P_cont = Float64(data["P_cont_W"])
    Nt = Int(data["Nt"])
    time_window = Float64(data["time_window_ps"])
    fiber_name = data["fiber_name"]

    # Reconstruct EXACT same grid (critical: Nt and time_window must match stored values)
    uω0, fiber, sim, band_mask, Δf, _ = setup_raman_problem(
        L_fiber=L, P_cont=P_cont, Nt=Nt, time_window=time_window,
        fiber_preset=preset
    )

    # Verify grid match
    @assert length(phi_opt) == size(uω0, 1) "Grid mismatch: phi_opt=$(length(phi_opt)) vs Nt=$(size(uω0,1))"

    # Set z-save points (D-01: 50 points, LinRange)
    fiber_shaped = deepcopy(fiber)
    fiber_shaped["zsave"] = collect(LinRange(0, fiber["L"], n_zsave))
    fiber_unshaped = deepcopy(fiber)
    fiber_unshaped["zsave"] = collect(LinRange(0, fiber["L"], n_zsave))

    # Shaped propagation (D-06: apply phi_opt)
    uω0_shaped = uω0 .* exp.(1im .* phi_opt)
    @info "Propagating shaped: $config_name (Nt=$Nt, L=$(L)m)"
    sol_shaped = MultiModeNoise.solve_disp_mmf(uω0_shaped, fiber_shaped, sim)

    # Unshaped propagation (D-06: flat phase)
    @info "Propagating unshaped: $config_name (Nt=$Nt, L=$(L)m)"
    sol_unshaped = MultiModeNoise.solve_disp_mmf(uω0, fiber_unshaped, sim)

    # J(z) at each z-slice (D-02)
    J_z_shaped   = [spectral_band_cost(sol_shaped["uω_z"][i, :, :],   band_mask)[1] for i in 1:n_zsave]
    J_z_unshaped = [spectral_band_cost(sol_unshaped["uω_z"][i, :, :], band_mask)[1] for i in 1:n_zsave]

    zsave = collect(LinRange(0, L, n_zsave))

    return (sol_shaped=sol_shaped, sol_unshaped=sol_unshaped,
            J_z_shaped=J_z_shaped, J_z_unshaped=J_z_unshaped,
            zsave=zsave, fiber_shaped=fiber_shaped, fiber_unshaped=fiber_unshaped,
            sim=sim, band_mask=band_mask, phi_opt=phi_opt, uω0=uω0,
            L=L, P_cont=P_cont, Nt=Nt, fiber_name=fiber_name, data=data)
end
```

**Helper function: `pz_save_to_jld2`** (per D-14)
Save each propagation pair immediately after computation to release memory (Research Pitfall 2):
```julia
function pz_save_to_jld2(result, fiber_dir, config_name)
    for (suffix, sol, J_z) in [
        ("shaped",   result.sol_shaped,   result.J_z_shaped),
        ("unshaped", result.sol_unshaped, result.J_z_unshaped)
    ]
        fname = "$(fiber_dir)_$(config_name)_$(suffix)_zsolved.jld2"
        fpath = joinpath(PZ_RESULTS_DIR, fname)
        JLD2.save(fpath,
            "uω_z",       sol["uω_z"],
            "ut_z",        sol["ut_z"],
            "J_z",         J_z,
            "zsave",       result.zsave,
            "phi_opt",     result.phi_opt,
            "L_m",         result.L,
            "P_cont_W",    result.P_cont,
            "fiber_name",  result.fiber_name,
            "Nt",          result.Nt,
            "sim_Dt",      result.sim["Δt"],
            "band_mask",   result.band_mask
        )
        @info "Saved $fpath"
    end
end
```
