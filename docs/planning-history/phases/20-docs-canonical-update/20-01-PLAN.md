```bash

  pdflatex -interaction=nonstopmode "$f.tex" > "/tmp/pdflatex-$f-pass1.log" 2>&1

  pdflatex -interaction=nonstopmode "$f.tex" > "/tmp/pdflatex-$f-pass2.log" 2>&1

```

Then verify each PDF rebuilt cleanly:

```bash
cd /Users/ignaciojlizama/RiveraLab/fiber-raman-suppression
for f in companion_explainer physics_verification verification_document; do
  if [ "docs/$f.pdf" -nt "docs/$f.tex" ]; then
    echo "$f.pdf: NEWER than .tex (good)"
  else
    echo "$f.pdf: STALE — pdflatex did not regenerate; investigate"
    exit 1
  fi
  if grep -E "^! " "docs/$f.log" > /dev/null; then
    echo "$f: LATEX ERROR — see docs/$f.log"
    grep -E "^! " "docs/$f.log" | head -5
    exit 1
  else
    echo "$f: clean build"
  fi
done
```
