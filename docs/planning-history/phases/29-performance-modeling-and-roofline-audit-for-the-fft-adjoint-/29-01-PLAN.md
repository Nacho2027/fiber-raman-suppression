<objective>
Lock the Phase 29 performance-modeling contract and deliver the execution-ready benchmark + roofline
protocol promised by the Phase 27 NMDS recommendation and the `performance-modeling-and-roofline-audit`
seed. Produce a small, stable benchmark suite that measures the forward/adjoint FFT pipeline at the
kernel level AND the full-solve level, a pure analysis library that turns those timings into
MEMORY_BOUND / COMPUTE_BOUND / SERIAL_BOUND verdicts against a captured hardware roofline, and a
modeled performance memo with an Executive Verdict naming the dominant bottleneck.

Purpose: Stop guessing where time goes. Before any further optimizer work, hardware spend, or
micro-optimization, Rivera Lab needs a defensible model of what `-t N` buys, where the FFT ridge
is, and how much of each wall-second is serial orchestration that no thread count can help.

Output: Five scripts (`bench_kernels.jl`, `bench_solves.jl`,
`bench_solves_run.jl`, `roofline_model.jl`, `report.jl`), one unit-test
file (`test/test_phase29_roofline.jl`), a raw-timing artifact set under `results/phase29/`, and a

    ```julia
    """
    Phase 29 subprocess worker. Prints exactly one `BENCH_JSON: {...}` line on
    stdout (everything else is discarded by the parent driver). The three modes
    measure DISTINCT work to defeat the "subtract forward from full recovers ≈ 0"
    failure mode:

      mode=="forward"  — direct forward-only solve via MultiModeNoise.solve_disp_mmf
                         (no cost, no adjoint). See 29-RESEARCH.md §7.1.
      mode=="adjoint"  — pre-run one forward OUTSIDE the timed block to capture
                         the ODESolution ũω and the terminal adjoint condition
                         λωL = spectral_band_cost(uωf, band_mask)[2], then time
                         only solve_adjoint_disp_mmf(λωL, ũω, fiber, sim).
                         See 29-RESEARCH.md §7.2.
      mode=="full_cg"  — time one call of cost_and_gradient (forward + cost +
                         adjoint + chain rule + regularizers + log-dB). See
                         29-RESEARCH.md §7.3.

    Consistency check done by the outer driver (29-RESEARCH.md §7.4):
      residual = median(full_cg) - (median(forward) + median(adjoint))
    The residual is the cost-accumulation + chain-rule + regularizer overhead
    (should be ≤ 10% of full_cg). Written to solves.jld2.
    """
    using Printf
    using JSON3
    using LinearAlgebra
    using FFTW
    using Random
    ENV["MPLBACKEND"] = "Agg"
    using MultiModeNoise
    using Tullio
    using Optim

    include(joinpath(@__DIR__, "common.jl"))
    include(joinpath(@__DIR__, "raman_optimization.jl"))

    # Guard so `julia -e 'include("bench_solves_run.jl")'` (parse-check)
    # does not crash on missing ARGS. Without this, the verify step's
    # `include(...)` linting will error on `ARGS[1]`.
    if isempty(ARGS)
        @info "phase29 worker: loaded via include with no ARGS — skipping timed block (parse-check mode)"
        return   # inside a `@main`/top-level return is a no-op; use `exit(0)` if preferred
    end

    mode = ARGS[1]    # "forward" | "adjoint" | "full_cg"
    tag  = ARGS[2]

    const W_NT       = 2^13
    const W_L_FIBER  = 2.0     # SMF-28 canonical
    const W_P_CONT   = 0.2

    Random.seed!(42)
    uω0, fiber, sim, band_mask, Δf, raman_threshold = setup_raman_problem(;
        L_fiber=W_L_FIBER, P_cont=W_P_CONT, Nt=W_NT,
        fiber_preset=:SMF28, β_order=3, time_window=10.0)
    Nt = sim["Nt"]; M = sim["M"]
    φ         = zeros(Nt, M)
    uω0_shaped = @. uω0 * cis(φ)    # mirror cost_and_gradient's production path

    # ── JIT warmup — compile every code path we will measure below. Do it for
    # all three paths so no branch pays a cold-start cost inside the timed block.
    _warm_fwd = MultiModeNoise.solve_disp_mmf(uω0_shaped, fiber, sim)
    let ũω = _warm_fwd["ode_sol"]
        uωf_warm = @. cis(fiber["Dω"] * fiber["L"]) * ũω(fiber["L"])
        _, λωL_warm = spectral_band_cost(uωf_warm, band_mask)
        _ = MultiModeNoise.solve_adjoint_disp_mmf(λωL_warm, ũω, fiber, sim)
    end
    _ = cost_and_gradient(φ, uω0, fiber, sim, band_mask)

    # ── Timed block. Each branch measures a DIFFERENT amount of work. ──
    elapsed_s = NaN
    J         = NaN
    iters     = 0

    if mode == "forward"
        # Pure forward propagation — no cost, no adjoint.
        # Source: src/simulation/simulate_disp_mmf.jl:178-200 (solve_disp_mmf).
        t_start = time()
        sol_fwd = MultiModeNoise.solve_disp_mmf(uω0_shaped, fiber, sim)
        elapsed_s = time() - t_start
        iters = length(sol_fwd["ode_sol"].t)   # accepted Tsit5 steps
        J = NaN                                  # no cost measured in this mode

    elseif mode == "adjoint"
        # Pre-capture forward ODE solution + terminal adjoint condition OUTSIDE
        # the timed block, then time ONLY the adjoint propagation. This isolates
        # adjoint cost without relying on post-hoc subtraction.
        # Source: src/simulation/sensitivity_disp_mmf.jl:294-304 (solve_adjoint_disp_mmf).
        sol_fwd_setup = MultiModeNoise.solve_disp_mmf(uω0_shaped, fiber, sim)
        ũω            = sol_fwd_setup["ode_sol"]
        L             = fiber["L"]
        Dω            = fiber["Dω"]
        uωf           = @. cis(Dω * L) * ũω(L)
        _, λωL        = spectral_band_cost(uωf, band_mask)

        t_start = time()
        sol_adj = MultiModeNoise.solve_adjoint_disp_mmf(λωL, ũω, fiber, sim)
        elapsed_s = time() - t_start
        # Touch the result to prevent DCE but outside the timed block.
        _λ0 = sol_adj(0)
        @assert all(isfinite, _λ0) "adjoint solution contains non-finite entries"
        iters = length(sol_adj.t)
        J = NaN

    elseif mode == "full_cg"
        # Full forward + cost + adjoint + chain rule (+ optional regularizers +
        # log-dB conversion). Source: scripts/raman_optimization.jl:73-178.
        t_start = time()
        J_val, _grad = cost_and_gradient(φ, uω0, fiber, sim, band_mask)
        elapsed_s = time() - t_start
        J = J_val
        iters = 1

    else
        error("Unknown mode: $mode (expected \"forward\" | \"adjoint\" | \"full_cg\")")
    end

    payload = Dict(
        "mode" => mode, "tag" => tag,
        "elapsed_s" => elapsed_s, "J" => J, "iters" => iters,
        "Nt" => Nt, "M" => M,
        "julia_threads" => Threads.nthreads(),
        "fftw_threads" => FFTW.get_num_threads(),
        "blas_threads" => BLAS.get_num_threads(),
    )
    println("BENCH_JSON: ", JSON3.write(payload))
    flush(stdout)
    ```

    The three modes measure DISTINCT quantities — this is non-negotiable. The
    prior draft had all three calling `cost_and_gradient`, which made the
    "subtract forward from full" trick recover ≈ 0. The correct call-path map
    (also in 29-RESEARCH.md §7):

    | mode       | Julia call                                                                 | source file:line                        |
    |------------|-----------------------------------------------------------------------------|-----------------------------------------|
    | forward    | `MultiModeNoise.solve_disp_mmf(uω0_shaped, fiber, sim)`                     | src/simulation/simulate_disp_mmf.jl:178 |
    | adjoint    | `MultiModeNoise.solve_adjoint_disp_mmf(λωL, ũω, fiber, sim)` (timed ONLY)   | src/simulation/sensitivity_disp_mmf.jl:294 |
    | full_cg    | `cost_and_gradient(φ, uω0, fiber, sim, band_mask)`                          | scripts/raman_optimization.jl:73        |

    Adjoint mode pre-captures `ũω` (forward ODESolution) and `λωL` (terminal
    condition from `spectral_band_cost(uωf, band_mask)` at scripts/common.jl:261)
    OUTSIDE the timed block — only the adjoint ODE solve is timed.

    # ═══════════════════════════════════════════════════════════════════════════════
    # Phase 29 Roofline Model — arithmetic intensity, bandwidth, Amdahl/Gustafson
    # ═══════════════════════════════════════════════════════════════════════════════
    # READ-ONLY consumer of timing JLD2 artifacts. No simulations, no ODE calls.
    # Constants use the P29M_ prefix (Phase 29 Modeling).
    #
    # Public API:
    #   arithmetic_intensity(flops, bytes) -> Float64 (FLOP/byte)
    #   roofline_bound(AI, peak_flops, peak_bw) -> NamedTuple(bound, regime)
    #   fit_amdahl(n_threads, times) -> NamedTuple(p, speedup_inf, rmse)
    #   fit_gustafson(n_threads, times_per_n) -> NamedTuple(s, speedup_n)
    #   kernel_regime_verdict(AI, bound_regime) -> String
    #   assemble_roofline_memo(bench_data::Dict; hw_profile::Dict) -> String (markdown)
    # ═══════════════════════════════════════════════════════════════════════════════

    using LinearAlgebra, Statistics, Printf

    if !(@isdefined _PHASE29_ROOFLINE_LOADED)

    const _PHASE29_ROOFLINE_LOADED = true
    const P29M_VERSION = "1.0.0"
    const P29M_MEMORY_BOUND_AI_THRESHOLD = 1.0   # FLOP/byte
    const P29M_COMPUTE_BOUND_AI_THRESHOLD = 10.0
    const P29M_SERIAL_FRACTION_SATURATION = 0.2  # p>0.2 means Amdahl saturates fast
    const P29M_REGIME_RANK = Dict("MEMORY_BOUND"=>0, "COMPUTE_BOUND"=>1, "SERIAL_BOUND"=>2, "AMDAHL_SATURATED"=>3, "UNKNOWN"=>4)

    """
        arithmetic_intensity(flops::Real, bytes::Real) -> Float64

    FLOP/byte for a kernel. Preconditions: `flops ≥ 0`, `bytes > 0`.
    """
    function arithmetic_intensity(flops::Real, bytes::Real)
        @assert flops >= 0 "flops must be non-negative, got $flops"
        @assert bytes > 0 "bytes must be positive, got $bytes"
        return Float64(flops) / Float64(bytes)
    end

    """
        roofline_bound(AI::Real, peak_flops::Real, peak_bw::Real) -> (bound_flops_s, regime)

    Classical roofline: min(peak_flops, peak_bw * AI). `regime` is "COMPUTE_BOUND" if the
    compute ceiling dominates, "MEMORY_BOUND" otherwise.
    """
    function roofline_bound(AI::Real, peak_flops::Real, peak_bw::Real)
        @assert AI >= 0 "AI must be non-negative"
        @assert peak_flops > 0 && peak_bw > 0 "peaks must be positive"
        mem_ceiling = peak_bw * AI
        if mem_ceiling >= peak_flops
            return (bound_flops_s = peak_flops, regime = "COMPUTE_BOUND")
        else
            return (bound_flops_s = mem_ceiling, regime = "MEMORY_BOUND")
        end
    end

    """
        fit_amdahl(n_threads::AbstractVector, times::AbstractVector) -> (p, speedup_inf, rmse)

    Amdahl model: T(n) = T(1) * ((1-p) + p/n) WHERE p is the PARALLEL fraction.
    Note convention: here `p` is the parallelizable fraction (so 1-p is serial);
    `speedup_inf = 1/(1-p)` is the asymptotic speedup ceiling. Uses least-squares fit in
    (1/n) space. Preconditions: n_threads ≥ 1, monotone increasing; times > 0; same length.
    """
    function fit_amdahl(n_threads::AbstractVector, times::AbstractVector)
        @assert length(n_threads) == length(times) "length mismatch"
        @assert all(n_threads .>= 1) "n_threads must be ≥ 1"
        @assert all(times .> 0) "times must be positive"
        T1 = Float64(times[1])
        # Solve min_p ||T_n - T1*((1-p) + p/n)||^2
        # Let x_n = 1/n, then T_n/T1 = (1-p) + p*x_n = 1 + p*(x_n - 1)
        # y_n = T_n/T1 - 1 = p*(x_n - 1); linear in p
        xs = 1.0 ./ Float64.(n_threads)
        ys = Float64.(times) ./ T1 .- 1.0
        A = xs .- 1.0
        p = sum(A .* ys) / sum(A .* A)
        p = clamp(p, 0.0, 1.0)
        pred = T1 .* ((1 - p) .+ p .* xs)
        rmse = sqrt(mean((Float64.(times) .- pred).^2))
        speedup_inf = p < 1.0 ? 1.0 / (1.0 - p) : Inf
        return (p = p, speedup_inf = speedup_inf, rmse = rmse)
    end

    """
        fit_gustafson(n_threads, times_fixed_work_per_thread) -> (s, speedup_n)

    Scaled-speedup fit where each thread keeps per-thread work constant. Returns serial
    fraction `s` and speedup at max n. Not always applicable (we measure fixed total work
    here); exported for completeness.
    """
    function fit_gustafson(n_threads::AbstractVector, times::AbstractVector)
        @assert length(n_threads) == length(times) "length mismatch"
        ns = Float64.(n_threads); ts = Float64.(times)
        # Gustafson: S(n) = n - s*(n-1). Fit s by least squares with S(n) = t_1/t_n * n.
        T1 = ts[1]
        S = T1 ./ ts .* ns    # naive; caller has responsibility for Gustafson assumption
        A = 1.0 .- ns
        y = S .- ns
        s = sum(A .* y) / sum(A .* A)
        s = clamp(s, 0.0, 1.0)
        speedup_n = maximum(S)
        return (s = s, speedup_n = speedup_n)
    end

    """
        kernel_regime_verdict(AI::Real, bound_regime::AbstractString) -> String

    One of "MEMORY_BOUND", "COMPUTE_BOUND", "SERIAL_BOUND", "AMDAHL_SATURATED", "UNKNOWN".
    Preconditions: bound_regime ∈ {"MEMORY_BOUND","COMPUTE_BOUND","UNKNOWN"}.
    """
    function kernel_regime_verdict(AI::Real, bound_regime::AbstractString)
        @assert AI >= 0 "AI non-negative"
        bound_regime in ("MEMORY_BOUND", "COMPUTE_BOUND", "UNKNOWN") ||
            throw(ArgumentError("unsupported bound_regime=$bound_regime"))
        AI < P29M_MEMORY_BOUND_AI_THRESHOLD && return "MEMORY_BOUND"
        AI > P29M_COMPUTE_BOUND_AI_THRESHOLD && return "COMPUTE_BOUND"
        return bound_regime  # intermediate region — trust the roofline
    end

    """
        assemble_roofline_memo(bench_data::Dict; hw_profile::Dict) -> String

    Render the full markdown memo used by scripts/report.jl. MUST include the
    headings: "# Phase 29 Report", "## Executive Verdict", "## Kernel Timings", "## Amdahl Fits",
    "## Roofline Regimes", "## Recommendations". Returns the markdown string (no I/O).
    """
    function assemble_roofline_memo(bench_data::Dict; hw_profile::Dict)
        io = IOBuffer()
        println(io, "# Phase 29 Report — Performance Modeling and Roofline Audit")
        println(io)
        println(io, "**Generated:** ", get(bench_data, "timestamp", "UNKNOWN"))
        println(io, "**Host:** ", get(hw_profile, "hostname", "UNKNOWN"),
                "  |  CPU: ", get(hw_profile, "cpu_info", "UNKNOWN"),
                "  |  Julia threads launched: ", get(hw_profile, "julia_threads", "?"))
        println(io)
        println(io, "## Executive Verdict")
        println(io, get(bench_data, "executive_verdict", "TODO"))
        println(io)
        println(io, "## Kernel Timings")
        # Caller fills the table; stub section here — real body assembled by report.jl
        println(io, get(bench_data, "kernels_md", "TODO"))
        println(io)
        println(io, "## Amdahl Fits")
        println(io, get(bench_data, "amdahl_md", "TODO"))
        println(io)
        println(io, "## Roofline Regimes")
        println(io, get(bench_data, "roofline_md", "TODO"))
        println(io)
        println(io, "## Recommendations")
        println(io, get(bench_data, "recommendations_md", "TODO"))
        return String(take!(io))
    end

    end # include guard

    function _executive_verdict(kernel_results, amdahl_fits)
        # One paragraph — identifies dominant bottleneck. Simple heuristic:
        # if any Amdahl p < 0.5 -> SERIAL_BOUND dominates;
        # else if FFT is on the memory-bound side of roofline -> MEMORY_BOUND dominates;
        # else COMPUTE_BOUND (unlikely in this codebase).
        min_p = minimum(get(fit, "p", 1.0) for (_, fit) in amdahl_fits)
        dominant = min_p < 0.5 ? "SERIAL_BOUND (orchestration + single-threaded RHS dominate)" :
                   min_p < 0.8 ? "MIXED (significant parallelizable fraction but serial tail matters)" :
                   "PARALLEL_SCALING (the pipeline scales nearly linearly up to the tested thread ceiling)"
        ceiling = min_p < 1.0 ? 1.0 / (1.0 - min_p) : Inf
        return @sprintf(
            "Dominant bottleneck: **%s**. Measured minimum parallelizable fraction across forward/adjoint/full_cg is p = %.3f, giving Amdahl speedup ceiling ≈ %.1fx. Recommendation: %s.",
            dominant, min_p, ceiling,
            min_p < 0.5 ? "do not pay for more than 4 burst-VM threads for the canonical single-mode workload; invest tuning effort in the FFT plan and per-RHS allocation path instead" :
            min_p < 0.8 ? "use -t 8 on the burst VM; going to -t 22 yields ≤ 1.5x over -t 8 at the measured p" :
            "use -t 22 on the burst VM; near-linear scaling is observed up to the tested ceiling")
    end

    function _kernels_md(results_table)
        io = IOBuffer()
        println(io, "| Kernel | n_fftw | n_runs | median (s) | throughput GB/s | notes |")
        println(io, "|--------|--------|--------|------------|-----------------|-------|")
        for (k, e) in results_table
            @printf(io, "| %s | %d | %d | %.4f | %s | %s |\n",
                get(e, :kernel, k),
                get(e, :n_fftw_threads, -1),
                get(e, :reps_per_block, -1),
                get(e, :time_median_s, NaN),
                haskey(e, :throughput_gb_s) ? @sprintf("%.2f", e.throughput_gb_s) : "—",
                get(e, :notes, ""))
        end
        return String(take!(io))
    end

    function _amdahl_md(amdahl_fits)
        io = IOBuffer()
        println(io, "| Mode | Fitted p | Speedup ceiling | RMSE (s) |")
        println(io, "|------|----------|------------------|----------|")
        for (mode, fit) in amdahl_fits
            @printf(io, "| %s | %.3f | %s | %.4e |\n", mode, fit["p"],
                isfinite(fit["speedup_inf"]) ? @sprintf("%.1fx", fit["speedup_inf"]) : "∞",
                fit["rmse"])
        end
        return String(take!(io))
    end

    function _roofline_md(results_table, hw_profile)

        io = IOBuffer()
        println(io, "| Kernel | AI (FLOP/byte) | Regime | Measured throughput | Ceiling | Utilization |")
        println(io, "|--------|----------------|--------|----------------------|---------|-------------|")
        # Calling code fills this when AI/ceilings are computed; skeleton shown.
        println(io, "<!-- Populated by analysis pass in report.jl main body -->")
        return String(take!(io))
    end

    function _recommendations_md(kernel_results, amdahl_fits, hw_profile)
        io = IOBuffer()
        println(io, "1. **FFT tuning effort**: <populated based on measured MEASURE vs ESTIMATE delta>")
        println(io, "2. **Thread count for production**: <populated based on fit_amdahl verdict>")
        println(io, "3. **Burst-VM economics**: <populated — cost/benefit of c3-highcpu-22 given measured scaling>")
        println(io, "4. **Next tuning target**: <kernel with worst roofline utilization, not lowest wall time>")
        return String(take!(io))
    end

    if abspath(PROGRAM_FILE) == @__FILE__
        isfile(P29R_KERNELS_JLD2) || error("missing $(P29R_KERNELS_JLD2) — run bench_kernels.jl first")
        isfile(P29R_SOLVES_JLD2)  || error("missing $(P29R_SOLVES_JLD2) — run bench_solves.jl first")

        kernel_payload = JLD2.load(P29R_KERNELS_JLD2)
        results_table  = kernel_payload["results_table"]

        bench = Dict(
            "timestamp"            => string(now()),
            "executive_verdict"    => _executive_verdict(results_table, amdahl_fits),
            "kernels_md"           => _kernels_md(results_table),
