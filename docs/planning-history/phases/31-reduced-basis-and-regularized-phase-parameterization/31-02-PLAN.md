    **Step 1: Add a Phase-31 penalty-aware wrapper** around `cost_and_gradient` at the top of the driver (after includes, before `run_branch_A`):

    ```julia
    """
        phase31_cost_and_gradient(phi, uω0, fiber, sim, band_mask; kwargs...)

    Extends cost_and_gradient with Phase 31 penalty kwargs. Applies penalties BEFORE log_cost rescale.
    Returns (J_total, grad_total, breakdown::Dict{String,Float64}).
    """
    function phase31_cost_and_gradient(phi::AbstractMatrix{<:Real}, uω0, fiber, sim,
                                       band_mask::AbstractVector{Bool};
                                       λ_tikhonov::Real = 0.0,
                                       λ_tod::Real = 0.0,
                                       λ_tv::Real = 0.0,
                                       λ_dct_l1::Real = 0.0,
                                       bw_mask::Union{Nothing,AbstractVector{Bool}} = nothing,
                                       B_dct_for_l1::Union{Nothing,AbstractMatrix{<:Real}} = nothing,
                                       log_cost::Bool = true,
                                       kwargs...)
        # Compute base cost + gradient WITHOUT log_cost (we apply it at the end together with penalties)
        J_raman_lin, grad_raman = cost_and_gradient(phi, uω0, fiber, sim, band_mask;
                                                   log_cost=false, kwargs...)
        J_total_ref = Ref(J_raman_lin)
        grad_total = copy(grad_raman)
        breakdown = Dict{String,Float64}("J_raman" => J_raman_lin,
                                         "J_tikhonov" => 0.0, "J_tod" => 0.0,
                                         "J_tv" => 0.0, "J_dct_l1" => 0.0)
        bw = bw_mask === nothing ? pulse_bandwidth_mask(uω0) : bw_mask
        breakdown["J_tikhonov"] = apply_tikhonov_phi!(J_total_ref, grad_total, phi, bw; λ=λ_tikhonov)
        breakdown["J_tod"]      = apply_tod_curvature!(J_total_ref, grad_total, phi, bw; λ=λ_tod, sim=sim)
        breakdown["J_tv"]       = apply_tv_phi!(J_total_ref, grad_total, phi, bw; λ=λ_tv)
        if λ_dct_l1 > 0 && B_dct_for_l1 !== nothing
            breakdown["J_dct_l1"] = apply_dct_l1!(J_total_ref, grad_total, phi, bw; λ=λ_dct_l1, B_dct=B_dct_for_l1)
        end
        J_total = J_total_ref[]
        if log_cost
            J_clamped = max(J_total, 1e-15)
            log_scale = 10.0 / (J_clamped * log(10.0))
            grad_total .*= log_scale
            J_total = 10.0 * log10(J_clamped)
        end
        @assert isfinite(J_total)
        @assert all(isfinite, grad_total)
        return J_total, grad_total, breakdown
    end
    ```

        setup_raman_problem_perturbed(; base_config, fwhm_scale=1.0, P_scale=1.0, β2_scale=1.0, Nt, time_window)

    is scaled relative to base_config. Implemented by calling setup_raman_problem with scaled kwargs —

    common.jl:339), both confirmed real.
    """
    function setup_raman_problem_perturbed(; base_config, fwhm_scale::Real=1.0, P_scale::Real=1.0,
                                           β2_scale::Real=1.0, Nt::Int=P31_NT, time_window::Real=P31_TIME_WINDOW)
        @assert fwhm_scale > 0
        @assert P_scale > 0
        @assert β2_scale > 0
        # Resolve the base fiber presets FWHM and β₂ defaults. setup_raman_problem defaults are:
        #   pulse_fwhm = 185e-15, betas_user = [-2.6e-26] (see common.jl:334, 339).
        # If base_config overrides either, use the override; otherwise use the defaults.
        base_fwhm = get(Dict(pairs(base_config)), :pulse_fwhm, 185e-15)
        base_betas = get(Dict(pairs(base_config)), :betas_user, [-2.6e-26])
        @assert length(base_betas) >= 1 "base_betas must contain at least β₂"
        # Build scaled betas vector: scale β₂ (index 1), leave higher orders untouched.
        scaled_betas = copy(base_betas)
        scaled_betas[1] = base_betas[1] * β2_scale
        uω0, fiber, sim, band_mask, Δf, raman_threshold = setup_raman_problem(;
            fiber_preset = base_config.fiber_preset,
            L_fiber      = base_config.L_fiber,
            P_cont       = base_config.P_cont * P_scale,
            pulse_fwhm   = base_fwhm * fwhm_scale,
            betas_user   = scaled_betas,
            Nt           = Nt,
            time_window  = time_window)
        # Both perturbations applied in-line via setup_raman_problem kwargs — no fallback flags needed.
        # The `fiber_preset` path may override `betas_user` inside setup_raman_problem (see common.jl:343,
        # `_apply_fiber_preset`). For fiber_preset != nothing, the caller is scaling a preset's β₂ — verify
        # that `_apply_fiber_preset` respects the user-supplied betas_user when fiber_preset is also given.
        # If the preset overrides our scaled β₂, record `"beta2_perturbation_applied" = false` in the row;
        # otherwise `true`. Check via: `abs(fiber["betas"][1] - scaled_betas[1]) < 1e-35` (exact match).
        beta2_ok = haskey(fiber, "betas") && length(fiber["betas"]) >= 1 &&
                   abs(fiber["betas"][1] - scaled_betas[1]) < 1e-35
        if β2_scale != 1.0 && !beta2_ok

        end

            fwhm_perturbation_applied = (fwhm_scale != 1.0),
            P_perturbation_applied    = (P_scale != 1.0),

    function compute_sigma_3dB(phi_opt::AbstractVector, uω0, fiber, sim, band_mask;
                               σ_ladder=[0.01, 0.02, 0.05, 0.1, 0.2, 0.5], n_trials=20, rng=Random.GLOBAL_RNG)
        Nt = length(phi_opt)
        J_base, _ = cost_and_gradient(reshape(phi_opt, Nt, 1), uω0, fiber, sim, band_mask; log_cost=true)
        for σ in σ_ladder
            deltas = Float64[]
            for _ in 1:n_trials
                noise = σ .* randn(rng, Nt)
                J_pert, _ = cost_and_gradient(reshape(phi_opt .+ noise, Nt, 1), uω0, fiber, sim, band_mask; log_cost=true)
                push!(deltas, J_pert - J_base)  # positive = worse (less negative dB)
            end
            if mean(deltas) ≥ 3.0
                return σ
            end
        end
        return σ_ladder[end]  # didn't reach 3 dB — pulse is extremely robust
    end
